category
-
cat_ids varchar(5) PK
category varchar(50)

subcategory
-
subcat_ids varchar(8) PK
subcategory varchar(60)

cat_subcat
-
cat_ids varchar(5) PK  FK >- category.cat_ids
subcat_ids varchar(8) PK FK >- subcategory.subcat_ids

country_currency
-
country varchar(2) PK
currency varchar(3) PK

contacts
-
contact_id int PK
first_name varchar(60)
last_name varchar(60)
email varchar

campaign
-
cf_id int PK
contact_id int FK - contacts.contact_id
company_name varchar(100)
description varchar(200)
goal float
pledge float
outcome float
backers_count int
country varchar(2) FK >- country_currency.country
currency varchar(3) FK >- country_currency.currency
launch_date date
end_date date
cat_ids varchar(5) FK >- cat_subcat.cat_ids
subcat_ids varchar(8) FK >- cat_subcat.subcat_ids
